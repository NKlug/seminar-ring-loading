	\begin{frame}{Training data}
		\begin{definition}
			For $1 \leq i \leq N$, let $x_i: (\Omega, \cF) \rightarrow (\R^n, \cB(\R^n))$ be independent and identically distributed random variables with $x_i \sim P_{data}$.\\
			$P_{data}$ is the distribution of the training data over the sample space $\R^n$.\\
			The \emph{training data} is a realization $(x_1(\omega), x_2(\omega), \dots, x_N(\omega))$ of random variables $x_i$ for some $\omega \in \Omega$.
		\end{definition}
		We write $X_i \coloneqq x_i(\omega)$ and $(X_1, \dots, X_n)$.
	\end{frame}
